    <script>
    
    function submitamount()
{
	var u = document.getElementById ('units').value;
    
    bill = 0;
    if(u>0 && u<=100)
    {
    	bill = u * 1;
    }
    if (u>100 && u<=200)
    {
    	bill = ((u-100) * 2) + (1*100);        
    }
    if (u>200 && u<=300)
    {
    	bill = ((u-200) * 3) + (1*100)+(2*100);        
    }
    if (u>300)
    {
    	bill = bill + ((u-300) * 5) + (1*100)+(2*100)+(3*100);     
    }
    
    document.getElementById('result').innerHTML = "Bill amount: "+bill;
}
    
    </script>
